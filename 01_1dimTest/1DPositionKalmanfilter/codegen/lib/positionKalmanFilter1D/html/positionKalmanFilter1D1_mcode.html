 50% % Kalman Filter for Position Estimator from Acc + GPS
 51% % 
 52% % state_vector x_aposteriori_k = [pos vel]'
 53% % P_aposteriori_k = [P11 P12; P21 P22]
 54% % velocitiy_observe = vo
 55% % gps_update = u
 56% % acc = a
 57% % z = [pos_meas vel_meas]'
 58% % sigma = [sigma1 sigma2]'
 59% % q = [q(1) q(2)]'
 60% % dt
 61% % $Author: Damian Aregger $    $Date: 2012 $    $Revision: 1 $
 62% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%\
 63
 64    % process noise variance
 65    Q = [q(1)^2*dt 0; 0 q(2)^2*dt];
 66    % measurement noise variance
 67    %R_full = [sigma(1)^2 0; 0 sigma(2)^2];
 68    R_full = [sigma(1)^2 0; 0 sigma(2)^2];
 69    % system description
 70    A = [1 dt; 0 1];
 71    B = [0.5*dt*dt dt]'*acc;
 72    % decide if velocity can be observed
 73    H_full = [1 0; 0 1];
 74    if velocity_observe == 0
 75        H = H_full(1,:);
 76        R = R_full(1,1);
 77    else
 78        H = H_full;
 79        R = R_full;
 80    end
 81    %% Main Filter step
 82    %Step 1 prediction/a priori
 83    x_apriori = A*x_aposteriori_k+B;                 %project state ahead
 84    P_apriori = A*P_aposteriori_k*A'+Q;                    %project error covariance ahead
 85    %Step 2 correction/a posteriori
 86    if gps_update == 1
 87        %K = P_apriori*H'*inv(H*P_apriori*H'+R);              %compute kalman gain
 88        K = (P_apriori*H')/(H*P_apriori*H'+R);
 89        if velocity_observe == 0
 90            x_aposteriori = x_apriori+K*(z(1,1)-H*x_apriori);        %update estimate via z
 91            P_aposteriori = (eye(2)-K*H)*P_apriori;           %update error covariance
 92        else
 93            x_aposteriori = x_apriori+K*(z(:,1)-H(1:2,1:2)*x_apriori);        %update estimate via z
 94            P_aposteriori = (eye(2)-K*H)*P_apriori;           %update error covariance
 95        end
 96    else
 97       x_aposteriori = x_apriori;
 98       P_aposteriori = P_apriori;
 99    end

  575   static real32_T rt_hypotf_snf(real32_T u0, real32_T u1)
  576   {
  577     real32_T y;
  578     real32_T a;
  579     a = (real32_T)fabs(u0);
  580     y = (real32_T)fabs(u1);
  581     if (a < y) {
  582       a /= y;
  583       y *= (real32_T)sqrt(a * a + 1.0F);
  584     } else if (a > y) {
  585       y /= a;
  586       y = a * (real32_T)sqrt(y * y + 1.0F);
  587     } else if (rtIsNaNF(y)) {
  588     } else {
  589       y = a * 1.41421354F;
  590     }

  592     return y;
  593   }

  595   static real32_T rt_powf_snf(real32_T u0, real32_T u1)
  596   {
  597     real32_T y;
  598     real32_T f2;
  599     real32_T f3;
  600     if (rtIsNaNF(u0) || rtIsNaNF(u1)) {
  601       y = ((real32_T)rtNaN);
  602     } else {
  603       f2 = (real32_T)fabs(u0);
  604       f3 = (real32_T)fabs(u1);
  605       if (rtIsInfF(u1)) {
  606         if (f2 == 1.0F) {
  607           y = ((real32_T)rtNaN);
  608         } else if (f2 > 1.0F) {
  609           if (u1 > 0.0F) {
  610             y = ((real32_T)rtInf);
  611           } else {
  612             y = 0.0F;
  613           }
  614         } else if (u1 > 0.0F) {
  615           y = 0.0F;
  616         } else {
  617           y = ((real32_T)rtInf);
  618         }
  619       } else if (f3 == 0.0F) {
  620         y = 1.0F;
  621       } else if (f3 == 1.0F) {
  622         if (u1 > 0.0F) {
  623           y = u0;
  624         } else {
  625           y = 1.0F / u0;
  626         }
  627       } else if (u1 == 2.0F) {
  628         y = u0 * u0;
  629       } else if ((u1 == 0.5F) && (u0 >= 0.0F)) {
  630         y = (real32_T)sqrt(u0);
  631       } else if ((u0 < 0.0F) && (u1 > (real32_T)floor(u1))) {
  632         y = ((real32_T)rtNaN);
  633       } else {
  634         y = (real32_T)pow(u0, u1);
  635       }
  636     }
  637   
  638     return y;
